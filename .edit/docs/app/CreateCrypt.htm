CreateCrypt

Use the CreateCrypt component to encrypt or decrypt a string with a given key or create a hash of it.

 app.CreateCrypt( options );

Example - Encrypt / Decrypt

function OnStart()
{
	var lay = app.CreateLayout("linear", "FillXY,VCenter");

	edtText = app.CreateTextEdit("", 0.8, 0.3);
	edtText.SetHint("text");
	lay.AddChild(edtText);

	edtKey = app.CreateTextEdit("", 0.8, 0.3);
	edtKey.SetHint("key");
	lay.AddChild(edtKey);

	spnMethod = app.CreateSpinner("Encrypt,Decrypt");
	lay.AddChild(spn);

	var btn = app.CreateButton("GO");
	btn.SetOnTouch(btn_OnTouch);
	lay.AddChild(btn);

	app.AddLayout(lay);
}

function btn_OnTouch()
{
	var txt = edtText.GetText(), key = edtKey.GetText(), op = spnMethod.GetText();
	if(key == "") return;

	crypt = app.CreateCrypt();
	if(op == "Encrypt") prompt("Encrypted:", crypt.Encrypt(txt, key));
	else if(op == "Encrypt") prompt("Decrypted:", crypt.Decrypt(txt, key));
}

Example - Hash

function OnStart()
{
	var pass = prompt("Please enter password", "");
	if(pass == null) return;

	
	var crypt = app.CreateCrypt();
	var md5 = crypt.Hash( pass, "MD5" );
	
	var chk = app.LoadText("hash", "none");

	if(chk == "none") app.SaveText("hash", md5);
	else if(chk == md5) app.ShowPopup("Welcome!");
	else app.ShowPopup("Incorrect password!);
}
